最近有给大家在2020《应用生物信息学》这门课上介绍“转录组测序数据分析”专题，由于在课上着重讲了从fastq原始数据开始到差异分析以及通路富集分析的整个流程，代码细节没有过多介绍，而课件上的一些代码在步骤与步骤之间也不太连续，导致新入门的小可爱们在做作业时，有点伤脑筋，所以今天本小编就这个专题的作业，给大家再详细介绍一下代码流程～

本流程代码所用的数据就是本次课作业里的两个文件：表达矩阵文件RawCounts.csv；样本信息文件sampleInfo.csv。

代码运行环境：R

读入表达矩阵数据和样本信息

需注意：
表达矩阵文件要以矩阵形式读入，样本信息文件要以数据框格式读入。

datExpr<-as.matrix(read.csv("RawCounts.csv", header=T, row.names=1, sep=","))
datMeta<-as.data.frame(read.csv("sampleInfo.csv", header=T, row.names=1))
dim(datExpr)
# [1] 58780   113
dim(datMeta)
# [1] 113   5

共58780个探针，113个样本。

低表达基因过滤（只保留在>50%的样本中表达值>1的基因）

删除在所有样本中均未检测到表达的基因这一标准还是太松了，会导致后面寻找隐藏协变量这一步出现错误：
Number of significant surrogate variables is: 3 Iteration (out of 5 ):Error in density.default(x, adjust = adj) : 'x' contains missing values

错误原因：
没有过滤掉低表达量的基因

解决办法：
在低表达基因过滤这一步条件设置得太松了，可改成：只保留在>50%的样本中表达值>1的基因。

pres = apply(datExpr>1,1,sum) 
to_keep = pres > 0.5 * ncol(datExpr)
table(to_keep)
# to_keep
# FALSE  TRUE
# 37578 21202
datExpr = datExpr[to_keep,]

基因过滤之后，只保留了21202个探针。

离群样本过滤

library(WGCNA)
sdout <- 2; normadj <- (0.5+0.5*bicor(datExpr, use='pairwise.complete.obs'))^2 
netsummary <- fundamentalNetworkConcepts(normadj); ku <- netsummary$Connectivity; z.ku <- (ku-mean(ku))/sqrt(var(ku)) 
outliers <- (z.ku > mean(z.ku)+sdout*sd(z.ku))|(z.ku < mean(z.ku)-sdout*sd(z.ku)) 
print(paste("There are ",sum(outliers), "outliers samples based on a bicor distance sample network connectivity standard deviation above",sdout,sep="")); 
print(colnames(datExpr)[outliers])
print(table(outliers)) 
# [1] "There are 5outliers samples based on a bicor distance sample network connectivity standard deviation above2"
# [1] "donor9_ileum_repA"   "donor10_ileum_repB"  "donor19_ileum_repA"  "donor26_rectum_repA" "donor30_rectum_repA"
# outliers
# FALSE  TRUE
#   108     5
pdf("Outlier Removal.pdf") 
plot(z.ku, col = as.numeric(datMeta$disease.state), pch=19, main="Outlier detection", ylab="Network Connectivity (z score)") 
abline(h=-2, lty=2) 
dev.off()
datExpr = datExpr[,!outliers]
datMeta = datMeta[!outliers,]

可见，本步过滤掉了5个离群样本，还剩108个样本。画出的图Outlier Removal.pdf如下：

寻找隐藏协变量（Call Hidden Factors）

library(sva)
mod=model.matrix(~as.factor(datMeta$sex)+as.factor(datMeta$ancestry)+datMeta$age+as.factor(datMeta$disease.state), data=datMeta) 
n.sv=num.sv(datExpr,mod,method="be") 
n.sv 
# [1] 3
mod0=model.matrix(~as.factor(datMeta$sex)+as.factor(datMeta$ancestry)+datMeta$age, data=datMeta) 
svobj = sva(datExpr,mod,mod0,n.sv=n.sv)$sv 
datMeta$svobj=svobj
write.csv(datMeta,"datMeta_PVCA.csv", row.names=F)
datMeta<-read.csv("datMeta_PVCA.csv")

协变量校正 & 差异分析

由于协变量校正之后，表达值可能会出现负值，而使用DESeq2做差异分析的表达矩阵中是不允许出现负值的，所以我们协变量校正和差异分析一起做，在构建design矩阵时，直接将需要校正的协变量放进去一起构建，起到一个校正的作用。

library(DESeq2)
# 取整,函数round
input_data <-round(datExpr,digits = 0)
# 构建dds矩阵
dds <- DESeqDataSetFromMatrix(countData = datExpr, colData = datMeta, design = ~sex+ancestry+age+disease.state+svobj.1+svobj.2+svobj.3)
# 标准化
dds <- DESeq(dds)
# 差异分析结果
res <- results(dds, contrast=c("disease.state","Inflammatory Bowel Disease","Control"), alpha=0.05) 
# 查看结果的概括统计量,可以看到上下调基因的总体情况
summary(res)
# out of 21202 with nonzero total read count
# adjusted p-value < 0.05
# LFC > 0 (up)       : 340, 1.6%
# LFC < 0 (down)     : 170, 0.8%
# outliers [1]       : 0, 0%
# low counts [2]     : 823, 3.9%
# (mean count < 3)
# [1] see 'cooksCutoff' argument of ?results
# [2] see 'independentFiltering' argument of ?results

# 输出差异分析的结果
write.csv(res,"res.csv")

对res文件做一些手动处理

用Excel打开res文件，给第一列添加列名为”EnsemblID”；

然后将第一列值的小数位数去除（以便后面进行基因匹配）；

并且根据第一列进行去重（先将第一列排序，然后高亮显示并找到第一列的重复值，删除重复两行中的其中一整行。这里处理方法比较简单粗暴，其实处理不同探针对应同一个基因的情况，是在差异分析之前，并且是有科学的处理办法的，但是这里为了简化步骤，不展开介绍了）。

此处删除了2行。

结果可视化（火山图）

library(ggplot2)
library(ggrepel)
# 读入上一步得到的结果文件
resdata<-read.csv("res.csv")
resdata$significant <- as.factor(resdata$padj<0.05 & abs(resdata$log2FoldChange) > 1)
pdf("Volcano.pdf")
ggplot(data=resdata, aes(x=log2FoldChange, y =-log10(padj),color =significant)) +
geom_point() +
ylim(0, 8)+
scale_color_manual(values =c("black","red"))+
geom_hline(yintercept = log10(0.05),lty=4,lwd=0.6,alpha=0.8)+
geom_vline(xintercept = c(1,-1),lty=4,lwd=0.6,alpha=0.8)+
theme_bw()+
theme(panel.border = element_blank(),
panel.grid.major = element_blank(),
panel.grid.minor = element_blank(),
axis.line = element_line(colour = "black"))+
labs(title="Volcano Plot", x="log2 (fold change)",y="-log10 (padj)")+
theme(plot.title = element_text(hjust = 0.5))+
geom_text_repel(data=subset(resdata, -log10(padj) > 6), aes(label=EnsemblID),col="black",alpha =0.8)  #标记出padj<10E-06的基因
dev.off()

功能注释和通路富集

# BiocManager::install("GOSemSim")
# BiocManager::install("ggraph")
# BiocManager::install("polyclip")
# BiocManager::install("clusterProfiler"): http://yulab-smu.top/clusterProfiler-book/chapter12.html
library(clusterProfiler)
library(AnnotationDbi)
library(org.Hs.eg.db)
# 读入res_significant.csv文件
dat<-read.csv("res_significant.csv")
# 转换基因名
transID = bitr(dat$EnsemblID, fromType="ENSEMBL", toType=c("SYMBOL", "ENTREZID"), OrgDb="org.Hs.eg.db")
# GO_ALL 注释
ALL <- enrichGO(transID$ENTREZID,
    "org.Hs.eg.db",
    keyType="ENTREZID",
    ont="ALL",
    pvalueCutoff=0.01,
    pAdjustMethod="BH",
    qvalueCutoff=0.1
) #Cutoff是为后面画图设置的阈值，并且实际上这里的qvalueCutoff=0.1不是给qvalue设的，是给p.adjust设的（这是个bug!）
ALL <- setReadable(ALL, OrgDb=org.Hs.eg.db)
write.table(as.data.frame(ALL@result), file="GO_ALL.csv", sep=",", row.names=F)
# 可视化
pdf("GO_ALL.pdf", width=15, height=20)
barplot(ALL, color = "p.adjust", showCategory=20, drop=T, split="ONTOLOGY")+ facet_grid(ONTOLOGY ~ ., scales = "free") #柱状图，按p.adjust从小到大排列，在BP，MF，CC条目中各选取p.adjust(这个包只能按照p.adjust排序和筛选前20个条目，不能按照pvalue或qvalue)最小的前20个条目画图
dotplot(ALL, color = "p.adjust", showCategory=20, x=~-log(p.adjust), split="ONTOLOGY")+ facet_grid(ONTOLOGY ~ ., scales = "free") #气泡图
dev.off()

# KEGG 注释
kegg <- enrichKEGG(transID$ENTREZID, organism="hsa", keyType="kegg", pvalueCutoff=0.01, pAdjustMethod="BH", qvalueCutoff=0.1) #Cutoff是为后面画图设置的阈值，并且实际上这里的qvalueCutoff=0.1不是给qvalue设的，是给p.adjust设的（这是个bug!）
kegg <- setReadable(kegg, OrgDb=org.Hs.eg.db, keyType="ENTREZID")
write.table(as.data.frame(kegg@result), file="KEGG.csv", sep=",", row.names=F)
# 可视化
pdf("KEGG.pdf")
barplot(kegg, color = "p.adjust", showCategory=20, drop=T) #柱状图，按p.adjust从小到大排列，选取p.adjust(这个包只能按照p.adjust排序和筛选前20个条目，不能按照pvalue或qvalue)最小的前20个条目画图
dotplot(kegg, color = "p.adjust", showCategory=20, x=~-log(p.adjust)) #气泡图
dev.off()
